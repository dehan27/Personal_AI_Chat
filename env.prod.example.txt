# ── Django ───────────────────────────────────────────────
# 안전한 랜덤 문자열. 개발용과 반드시 다른 값으로.
# 생성 예: python -c "import secrets; print(secrets.token_urlsafe(50))"
DJANGO_SECRET_KEY=여기에_운영용_랜덤_키_입력

# 프로덕션은 False 필수.
DJANGO_DEBUG=False

# 외부 접속 허용 호스트. 쉼표로 여러 개 나열 가능. 스킴 없이 도메인만.
DJANGO_ALLOWED_HOSTS=여기에_운영_도메인_입력

# CSRF 신뢰 오리진. 반드시 스킴 포함(https://).
DJANGO_CSRF_TRUSTED_ORIGINS=https://여기에_운영_도메인_입력

# ── PostgreSQL ───────────────────────────────────────────
# 컨테이너 초기화 시점에 DB/사용자/비밀번호가 만들어짐. 이후 변경은 볼륨 재생성 필요.
POSTGRES_USER=여기에_DB_사용자명_입력
POSTGRES_PASSWORD=여기에_강한_DB_비밀번호_입력
POSTGRES_DB=여기에_DB_이름_입력

# ── OpenAI ───────────────────────────────────────────────
OPENAI_API_KEY=여기에_OpenAI_API_키_입력
OPENAI_MODEL=gpt-4o-mini

# ── 호스트 포트 매핑 ─────────────────────────────────────
# 실제 값은 운영 환경 상황에 맞춰 채워 넣음.
WEB_HOST_PORT=
DB_HOST_PORT=

# ── 배포할 이미지 태그 ────────────────────────────────────
# GitHub Actions 가 태그 푸시할 때 GHCR 에 같은 이름으로 올림.
# 예: v0.3.0, v0.3.1, latest
# Actions 가 SSH 로 배포할 때 이 값을 자동 치환해 준다.
APP_IMAGE_TAG=latest
