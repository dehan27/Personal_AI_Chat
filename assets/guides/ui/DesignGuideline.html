/* 2. 오류 표시 + 쉐이킹 + 스크롤 */
function showFieldError(input, msgEl, message) {
  input.classList.add('is-error');
  msgEl.textContent = message;
  msgEl.style.display = 'block';

  // 쉐이킹 — 이미 실행 중이면 재시작
  input.classList.remove('shake');
  requestAnimationFrame(() => input.classList.add('shake'));
  input.addEventListener('animationend', () => input.classList.remove('shake'), { once: true });
}

function clearFieldError(input, msgEl) {
  input.classList.remove('is-error');
  msgEl.style.display = 'none';
}

function validateAndSubmit(fields) {
  let firstError = null;

  fields.forEach(({ input, msgEl, validate, message }) => {
    if (!validate(input.value)) {
      showFieldError(input, msgEl, message);
      if (!firstError) firstError = input;
    } else {
      clearFieldError(input, msgEl);
    }
  });

  if (firstError) {
    // 첫 번째 오류 필드로 스크롤
    firstError.scrollIntoView({ behavior: 'smooth', block: 'center' });
    firstError.focus();
    return false;
  }
  return true;
}

/* Toast는 body에 .toast-container 추가 후 동적 생성 */
function showToast(type, title, msg) {
  const toast = document.createElement('div');
  toast.className = 'toast';
  toast.innerHTML = `
    <div class="toast-icon ${type}">...</div>
    <div class="toast-content">
      <div class="toast-title">${title}</div>
      <div class="toast-msg">${msg}</div>
    </div>
    <button class="toast-close" onclick="removeToast(this)">✕</button>
  `;
  document.querySelector('.toast-container').appendChild(toast);
  setTimeout(() => removeToast(toast.querySelector('.toast-close')), 4000);
}

제3조 (약관의 효력 및 변경)
본 약관은 서비스를 이용하고자 하는 모든 이용자에게 적용됩니다. 회사는 약관을 변경할 수 있으며, 변경된 약관은 공지 후 효력이 발생합니다.

제4조 (서비스의 제공)
회사는 서비스를 연중무휴, 1일 24시간 제공함을 원칙으로 합니다. 단, 시스템 정기점검 등 부득이한 사유가 있는 경우 서비스 제공을 일시 중단할 수 있습니다.

제5조 (이용자의 의무)
이용자는 관계 법령, 본 약관의 규정, 이용안내 및 서비스와 관련하여 공지한 주의사항, 회사가 통보하는 사항 등을 준수하여야 합니다.

제6조 (개인정보 보호)
회사는 이용자의 개인정보를 보호하기 위해 개인정보처리방침을 수립하고 이를 준수합니다.
